function shouldWater(tempC, airRH, soilMoist, lightLevel):

  SOIL_DRY = 30          # below this -> usually water
  SOIL_VERY_DRY = 20     # below this -> water even if "night"
  LIGHT_NIGHT = 15       # below this -> treat as night (tune to your scale)

  # --- Compute simple "demand score" from climate ---
  # Higher temp => more demand
  # Lower humidity => more demand
  # Higher light => more demand
  demand = 0

  if tempC > 32: demand += 1

  if airRH < 45: demand += 1
  if airRH < 35: demand += 1

  if lightLevel > 60: demand += 1
  if lightLevel > 80: demand += 1

  # Shift the soil threshold slightly based on demand (0..~6 points)
  soilThreshold = SOIL_DRY + 2 * demand

  # 1) Always water if very dry
  if soilMoist < SOIL_VERY_DRY:

  if (lightLevel < LIGHT_NIGHT) AND (soilMoist < SOIL_DRY):
    return (true, "night but soil dry")

  # 3) Daytime decision
  if soilMoist < soilThreshold:
    return (true, "soil below threshold (demand-adjusted)")

    return (false, "soil ok")
